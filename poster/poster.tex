% =============== Section 1: Introduction & Problem ===============
\begin{sectionbox}{1. What is Speech Enhancement?}

\textbf{Goal:} Extract the \emph{clean} speech from a noisy recording. Given a noisy signal $\mathbf{x}_\text{noisy} = \mathbf{x}_\text{clean} + \mathbf{n}$, recover $\mathbf{x}_\text{clean}$.

\textbf{Why synthetic data?} In practice, we never have a ``noisy + clean'' pair of the same utterance. The standard approach is to \textbf{synthesise} noisy data: take clean recordings (\textbf{LibriSpeech} dev-clean) and add pure noise (\textbf{MUSAN}) at controlled signal-to-noise ratios (SNR). This way, we always have the ground-truth clean speech for supervision.

\textbf{Data:} 2703 utterances, noise added at SNR $\in \{-5, 0, 5, 10, 15\}$\,dB. We train on \textbf{5\,dB} as a quick proof-of-concept. Split: 2163\,/\,270\,/\,270.

\textbf{Our approach:} Operate in the \emph{latent space} of two pre-trained audio tokenizers:
\begin{itemize}[leftmargin=10pt,itemsep=0pt,topsep=1pt]
  \item \textbf{DAC} [2] (Descript Audio Codec): compresses 16\,kHz audio to 50\,Hz latents ($d\!=\!1024$). We use it for both \textbf{source} (noisy) and \textbf{target} (clean) representations.
  \item \textbf{MOSS} [3] (OpenMOSS Audio Tokenizer): 4 hierarchical stages, 68 transformer layers, 12.5\,Hz output ($d\!=\!1280$). We use it to extract rich \textbf{condition} features from the noisy speech.
\end{itemize}

\begin{highlightbox}
\textbf{Key Idea:} Use \emph{Rectified Flow} [1] to learn a direct path from the noisy DAC latent ($\mathbf{x}_0$) to the clean DAC latent ($\mathbf{x}_1$), guided by multi-layer MOSS features as conditioning.
\end{highlightbox}

\end{sectionbox}

\vspace{-2pt}

% =============== Section 2a: Architecture (Preprocessing + Training) ===============
\begin{sectionbox}{2. Architecture --- Preprocessing \& Training}

\textbf{(a) Preprocessing} (offline, before training):
\begin{itemize}[leftmargin=10pt,itemsep=0pt,topsep=1pt]
  \item \textbf{DAC Encoder} [2] on clean speech $\to$ $\mathbf{x}_1 \in \mathbb{R}^{T \times 1024}$ (target latent)
  \item \textbf{DAC Encoder} [2] on noisy speech $\to$ $\mathbf{x}_0 \in \mathbb{R}^{T \times 1024}$ (source latent)
  \item \textbf{MOSS Tokenizer} [3] on noisy speech $\to$ 32 layer features $\to$ condition $\mathbf{c}$ (see \S3)
\end{itemize}

\textbf{(b) Training} (see Figure~1):
Sample $t\!\sim\!U(0,1)$. Interpolate: $\mathbf{x}_t = t\,\mathbf{x}_1 + (1\!-\!t)\,\mathbf{x}_0$.
A \textbf{Diffusion Transformer} (DiT [4]) with 6 layers predicts the velocity $\mathbf{v}_\theta(\mathbf{x}_t, t, \mathbf{c})$.
Each DiT layer: \textbf{AdaLN}$(t)$ $\to$ \textbf{Self-Attn} $\to$ \textbf{Cross-Attn}$(\mathbf{c}$ as K,V$)$ $\to$ \textbf{FFN}.
The MOSS condition $\mathbf{c}$ is injected via \textbf{Cross-Attention} (Q from latent, K/V from $\mathbf{c}$).

$\mathcal{L} = \|\mathbf{v}_\theta(\mathbf{x}_t,t,\mathbf{c}) - (\mathbf{x}_1 - \mathbf{x}_0)\|^2$

\end{sectionbox}

% =============== Section 2b: Architecture (Inference) ===============
\begin{sectionbox}{2. Architecture --- Inference (cont.)}

\textbf{(c) Inference} (see Figure~2):
No clean speech needed. Start from $\mathbf{x}_0$ (noisy DAC latent), run \textbf{50-step Euler ODE} [1] using the trained DiT, then decode $\hat{\mathbf{x}}_1$ with \textbf{DAC Decoder} [2] $\to$ enhanced speech.

% =============== Section 3: Research Question ===============
\begin{sectionbox}{3. Core Contribution: Multi-Layer Conditioning}

In standard practice, when using a pre-trained feature extractor as a conditioner, only the \textbf{last layer} output is used. But earlier layers also contain valuable information: early layers capture \textbf{fine-grained acoustic} details, while later layers encode more \textbf{abstract/semantic} features.

\begin{highlightbox}
\textbf{Research Question:} Can we improve speech enhancement by using \emph{all} 32 MOSS layers instead of just the last one? And can we let the model \emph{dynamically} choose which layers matter at each step of the denoising process?
\end{highlightbox}

This is analogous to multi-scale feature extraction in vision (FPN, U-Net). Our core innovation is a \textbf{time-dependent fusion} mechanism: a lightweight MLP maps ODE timestep $t$ to 32 softmax-normalised layer weights, so the model can emphasise different layers at different denoising stages.

% =============== Section 4: Experiment Design ===============
\begin{sectionbox}{4. Experiment Design}

\textbf{Four experiments} --- identical architecture (28.4M base), varying only the conditioning:

\begin{center}
\footnotesize
\begin{tabular}{clll}
\toprule
\textbf{Exp} & \textbf{Condition} & \textbf{MOSS Input} & \textbf{Extra Params} \\
\midrule
1 & No conditioning & --- & 28.4M (baseline) \\
2 & Last layer only & $\mathbf{h}_{31}$ & +10.1M cross-attn \\
3 & Static multi-layer & All 32 layers & +32 weights \\
4 & Time-dep.\ multi-layer & All 32 layers & +2,208 (MLP) \\
\bottomrule
\end{tabular}
\end{center}

\textbf{Conditioning variants:}
\begin{enumerate}[leftmargin=14pt,itemsep=1pt,topsep=1pt]
  \setcounter{enumi}{0}
  \item \textbf{None:} $\mathbf{c} = \varnothing$ (no MOSS features, no cross-attention layers)
  \item \textbf{Last Layer:} $\mathbf{c} = \mathbf{h}_{31}$
  \item \textbf{Static:} $\mathbf{c} = \sum_i \text{softmax}(\mathbf{w})_i \cdot \mathbf{h}_i$ \enspace (32 learnable scalar weights)
  \item \textbf{Time-Dep.:} $\mathbf{c}(t) = \sum_i \text{softmax}\!\big(\text{MLP}(t)\big)_i \cdot \mathbf{h}_i$
    \enspace (MLP: $1 \!\to\!64 \!\xrightarrow{\text{SiLU}}\! 32$)
\end{enumerate}

% Ablation figure (moved here from Section 3)
\begin{center}
\includegraphics[width=\linewidth]{image4_ablation.png}
\end{center}

\textbf{Training} (identical for all): AdamW, lr\,=\,$10^{-4}$, batch 16, max 100 epochs, cosine annealing, warmup 200 steps, early stopping (patience 20).

\smallskip
\textbf{Evaluation Metrics:}
\begin{itemize}[leftmargin=10pt,itemsep=1pt,topsep=1pt]
  \item \textbf{PESQ} (Perceptual Evaluation of Speech Quality, $\uparrow$): ITU-T P.862.
    Approximates human Mean Opinion Score via an auditory model.
    Range $[-0.5,\;4.5]$. Typical scores: noisy input $\sim$1.5--2.0, strong supervised models $\sim$2.5--3.5, clean reference\,=\,4.5.
  \item \textbf{STOI} (Short-Time Objective Intelligibility, $\uparrow$):
    $\text{STOI} = \frac{1}{TM}\sum_{t,m} \rho\!\left(\mathbf{x}_{t,m},\; \hat{\mathbf{x}}_{t,m}\right)$
    \enspace (average correlation of short-time spectral envelopes).
    Range $[0,\;1]$. Predicts human word-recognition rates. Typical: noisy $\sim$0.80, good models $\sim$0.90+.
  \item \textbf{FAD} (Fr\'echet Audio Distance, $\downarrow$): measures distributional similarity in VGGish embedding space.
    $\text{FAD} = \|\boldsymbol{\mu}_r - \boldsymbol{\mu}_g\|^2 + \text{Tr}\!\left(\boldsymbol{\Sigma}_r + \boldsymbol{\Sigma}_g - 2(\boldsymbol{\Sigma}_r\boldsymbol{\Sigma}_g)^{1/2}\right)$.
    Lower\,=\,more natural.
\end{itemize}

\textit{Note: PESQ scores from generative/flow-based models are typically lower than supervised discriminative baselines, as these models optimise perceptual naturalness rather than point-wise signal fidelity.}

\end{sectionbox}

% =============== Section 6: Limitations & Future Work ===============
\begin{sectionbox}{6. Limitations \& Future Work}

\textbf{Limitations:}
\begin{itemize}[leftmargin=10pt,itemsep=0pt,topsep=1pt]
  \item Trained on a single SNR (5\,dB) as proof-of-concept; data available for $-$5 to 15\,dB.
  \item Using all 32 MOSS layers means \textbf{32$\times$ more conditioning input} than single-layer baselines --- a trade-off between conditioning richness and computational cost.
\end{itemize}

\textbf{Future Work:}
\begin{itemize}[leftmargin=10pt,itemsep=0pt,topsep=1pt]
  \item \textbf{Layer selection study:} based on weight analysis, test fewer layers (e.g.\ first 15\,+\,last, or only first\,+\,last) to reduce the 32$\times$ cost while retaining multi-layer benefits.
  \item \textbf{Multi-SNR training} ($-$5 to 15\,dB) for a robust, general-purpose model.
  \item \textbf{Scale up:} larger datasets (full LibriSpeech + DNS Challenge), faster inference via ODE distillation.
\end{itemize}
