% =============== Section 1: Introduction & Problem ===============
\begin{sectionbox}{1. Speech Enhancement \& Motivation}

\textbf{Problem:} Recover clean speech $\mathbf{x}_\text{clean}$ from noisy $\mathbf{x}_\text{noisy} = \mathbf{x}_\text{clean} + \mathbf{n}$.

\textbf{Approach:} Operate in the \emph{latent space} of pre-trained audio tokenizers:
\begin{itemize}[leftmargin=10pt,itemsep=0pt,topsep=1pt]
  \item \textbf{DAC} (Descript Audio Codec): 16\,kHz $\to$ 50\,Hz latents, $\mathbf{z}\!\in\!\mathbb{R}^{1024}$ --- \textbf{target}
  \item \textbf{MOSS} (OpenMOSS): 4 hierarchical stages, 68 transformer layers, 12.5\,Hz --- \textbf{condition}
\end{itemize}

\begin{highlightbox}
\textbf{Key Idea:} Rectified Flow learns a direct ODE from noise to clean DAC latents, conditioned on multi-layer MOSS features from the noisy input.
\end{highlightbox}

\textbf{Data:} 2703 VCTK utterances + DEMAND noise (SNR\,=\,5\,dB). Split: 2163/270/270.

\end{sectionbox}

\vspace{-2pt}

% =============== Section 2: Model Architecture ===============
\begin{sectionbox}{2. Model Architecture}

\textbf{Diffusion Transformer (DiT)} --- 6 layers, 28.4M base params:

\begin{center}
\footnotesize
\begin{tabular}{ll}
\toprule
\textbf{Component} & \textbf{Spec} \\
\midrule
DAC latent / Hidden / MOSS dim & 1024 / 512 / 1280 \\
Heads / FFN dim & 8 / 2048 \\
Timestep embed & Sinusoidal $\to$ MLP \\
\bottomrule
\end{tabular}
\end{center}

\vspace{-2pt}
Each block: \textbf{AdaLN}(t) $\to$ \textbf{Self-Attn} $\to$ \textbf{Cross-Attn}(c) $\to$ \textbf{FFN}

\smallskip
\textbf{Rectified Flow:} \enspace
$\mathbf{x}_t = t\,\mathbf{x}_1 + (1\!-\!t)\,\mathbf{x}_0$, \enspace
$\mathcal{L} = \|\mathbf{v}_\theta(\mathbf{x}_t,t,\mathbf{c}) - (\mathbf{x}_1\!-\!\mathbf{x}_0)\|^2$

Inference: 50-step Euler from $\mathbf{x}_0\!\sim\!\mathcal{N}(0,\mathbf{I})$ to $\hat{\mathbf{x}}_1$, decoded by DAC.

% =============== Section 3: Research Question ===============
\begin{sectionbox}{3. Research Question: Multi-Layer Conditioning}

\textbf{MOSS Stage 3} has \textbf{32 transformer layers} ($d\!=\!1280$):
{\small early layers (0--7) capture fine-grained acoustics; middle (8--23) intermediate; late (24--31) semantic.}

\begin{highlightbox}
\textbf{Hypothesis:} A weighted combination of all 32 layers provides richer conditioning than using only the last layer --- analogous to multi-scale features in vision (FPN, U-Net).
\end{highlightbox}

\textbf{Three conditioning strategies:}
\begin{enumerate}[leftmargin=14pt,itemsep=1pt,topsep=1pt]
  \item \textbf{Last Layer:} $\mathbf{c} = \mathbf{h}_{31}$
  \item \textbf{Static Multi-Layer:} $\mathbf{c} = \sum_i \text{softmax}(\mathbf{w})_i \cdot \mathbf{h}_i$ \enspace (32 learnable weights)
  \item \textbf{Time-Dep.\ Multi-Layer:} $\mathbf{c}(t) = \sum_i \text{softmax}\!\big(\text{MLP}(t)\big)_i \cdot \mathbf{h}_i$\\
    MLP: $t \!\mapsto\! \mathbb{R}^{64} \!\xrightarrow{\text{SiLU}}\! \mathbb{R}^{32}$ \enspace (2,208 extra params)
\end{enumerate}

% =============== Section 4: Experiment Design ===============
\begin{sectionbox}{4. Experiment Design}

\textbf{Four experiments} --- identical architecture, varying only the conditioning:

\begin{center}
\footnotesize
\begin{tabular}{clll}
\toprule
\textbf{Exp} & \textbf{Condition} & \textbf{MOSS Input} & \textbf{Extra Params} \\
\midrule
1 & No conditioning & --- & 28.4M (baseline) \\
2 & Last layer only & $\mathbf{h}_{31}$ & +10.1M cross-attn \\
3 & Static multi-layer & All 32 layers & +32 weights \\
4 & Time-dep.\ multi-layer & All 32 layers & +2,208 (MLP) \\
\bottomrule
\end{tabular}
\end{center}

\textbf{Training} (identical for all): AdamW, lr\,=\,$10^{-4}$, batch 16, max 100 epochs, cosine annealing, warmup 200 steps, early stopping (patience 20).

\smallskip
\textbf{Evaluation Metrics:}
\begin{itemize}[leftmargin=10pt,itemsep=0pt,topsep=1pt]
  \item \textbf{PESQ} (Perceptual Evaluation of Speech Quality, $\uparrow$): ITU-T P.862 standard.\\
    Range $[-0.5,\;4.5]$. Compares degraded vs reference via auditory model. Higher\,=\,better.
  \item \textbf{STOI} (Short-Time Objective Intelligibility, $\uparrow$): Correlation of short-time envelopes.\\
    Range $[0,\;1]$. Predicts human intelligibility scores. Higher\,=\,better.
  \item \textbf{FAD} (Fr\'echet Audio Distance, $\downarrow$): Fr\'echet distance in VGGish embedding space.\\
    $\text{FAD} = \|\boldsymbol{\mu}_r - \boldsymbol{\mu}_g\|^2 + \text{Tr}\!\left(\boldsymbol{\Sigma}_r + \boldsymbol{\Sigma}_g - 2(\boldsymbol{\Sigma}_r\boldsymbol{\Sigma}_g)^{1/2}\right)$. \; Lower\,=\,more natural.
\end{itemize}

\end{sectionbox}

% =============== Section 6: Limitations & Future Work ===============
\begin{sectionbox}{6. Limitations \& Future Work}

\textbf{Limitations:} Single SNR (5\,dB) proof-of-concept; small dataset (2703 utterances); 50-step ODE slower than discriminative methods; moderate absolute PESQ ($\sim$1.70).

\smallskip
\textbf{Future:}
\begin{itemize}[leftmargin=10pt,itemsep=0pt,topsep=1pt]
  \item Multi-SNR training ($-$5 to 15\,dB) for robustness
  \item Few-layer selection (use only layers 0--15, 31 based on weight analysis)
  \item Larger data (full VCTK + DNS Challenge), faster inference (distillation)
\end{itemize}
